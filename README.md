**1. Introduction**
This project focuses on designing and implementing a realworld Hyperlocal Delivery Platform Database using Microsoft SQL Server. The system manages customers, vendors, products, orders, payments, delivery tracking, and reviews in a structured and scalable way. It simulates platforms like Foodpanda, Careem, or local delivery apps.


**2. Project Objective**
The main objective is to build a robust backend database that:
Handles customer & address management
Manages vendors and product inventory
Tracks orders and deliveries
Maintains payment lifecycle
Stores feedback and ratings
Supports rolebased access control


**3. Project Scope**
Supports multiple users and vendors
Allows multiple delivery addresses per user
Tracks complete order lifecycle
Supports scalability for future growth
Ensures data integrity using constraints

**4. Tools & Technologies Used**
DBMS: Microsoft SQL Server
Language: SQL (DDL, DML, DCL)
Concepts: Normalization, Keys, Triggers, Views, Indexes, Stored Procedures

**6. Database Design Overview**
The database is designed following industry best practices to ensure performance, security, and scalability.
Key Design Goals
Avoid data redundancy
Maintain referential integrity
Support future expansion
Enable efficient reporting

**7. Database Design Principles
7.1 Normalization**
Tables are normalized up to 3NF
Status values moved to a separate Status table
**7.2 Primary & Foreign Keys**
Each table has a unique Primary Key
Foreign Keys enforce relationships between entities
**7.3 Timestamps**
CreatedAt, UpdatedAt, PaidAt used for tracking
**7.4 AutoIncrement**
IDENTITY(1,1) used for all primary keys
**7.5 Scalability**
Separate tables for inventory, reviews, and status
Supports addition of new vendors, products, and users

    CREATEDAT DATETIME DEFAULT GETDATE()

CREATE TABLE Vendors (
    VENDORID INT PRIMARY KEY,
    VENDORNAME NVARCHAR(255),
    CONTACTMAIL NVARCHAR(100),
    CREATEDAT DATETIME DEFAULT GETDATE()

CREATE TABLE ProductCategories (
    CATEGORIESID INT PRIMARY KEY,
    CATEGORIESNAME NVARCHAR(100) NOT NULL,

--5.PRODUCTS TABLE 
CREATE TABLE Products(
    PRODUCTSID INT PRIMARY KEY IDENTITY(1,1),
    PRODUCTSNAME  NVARCHAR(100) NOT NULL,
    DESCRIPTION NVARCHAR(255),

12. Advanced SQL Features Used
12.1 Views
Used to simplify complex joins and reporting.
OrderSummary
ProductSales
CREATE VIEW OrderSummary AS
SELECT 
    O.ORDERSID,
    U.FULLNAME,
    V.VENDORNAME,
    O.TOTALAMOUNT,
    O.ORDERUPDATE
FROM Orders O
JOIN Users U ON O.USERID = U.USERID
JOIN Vendors V ON O.VENDORID = V.VENDORID;


SELECT * FROM OrderSummary;


--- ProductSales  

CREATE VIEW ProductSales AS
SELECT 
    P.PRODUCTSNAME,
    SUM(oi.QUANTITY) AS TotalSold
FROM OrdersItem oi
JOIN Products P ON oi.PRODUCTSID = P.PRODUCTSID
GROUP BY P.PRODUCTSNAME;


SELECT * FROM ProductSales;

ON OrdersItem
AFTER INSERT
AS
BEGIN
    UPDATE Inventory

    FROM Inventory inv
    JOIN inserted i ON inv.PRODUCTSID = i.PRODUCTSID;

SELECT 
    p.PRODUCTSNAME,
    AVG(r.RATING) AS AVERAGE_RATING
FROM Reviews r

--6.DELIVERY STATUS OF A PARTICULAR ORDER 
SELECT 
    O.ORDERSID,
    S.STATUSNAME,
    DS.UPDATEDAT
FROM DeliveryStatus DS
INNER JOIN Orders O
    ON DS.ORDERSID = O.ORDERSID
INNER JOIN Status S
    ON DS.STATUSID = S.STATUSID
WHERE O.ORDERSID = 5;   -- change to any order ID you want

--7.DALIY ORDER COUNT
SELECT 
    CAST(ORDERUPDATE AS DATE) AS ORDERDATE,
    COUNT(*) AS DAILY_ORDER_COUNT
FROM Orders
GROUP BY CAST(ORDERUPDATE AS DATE)
ORDER BY ORDERDATE ASC;

--8.TO CUSTOMERS BY ORDER VALUE 
SELECT 
    U.FULLNAME,
    SUM(O.TOTALAMOUNT) AS TOTAL_SPENT
FROM Orders O
INNER JOIN Users U
    ON O.USERID = U.USERID
GROUP BY U.FULLNAME
ORDER BY TOTAL_SPENT DESC;   -- highest spenders first
--9.LOW STOCK PRODUCTS PER VENDER
SELECT
    V.VENDORNAME,
    P.PRODUCTSNAME,
    I.QUANTITYAVAILABLE
FROM Inventory I
INNER JOIN Vendors V
    ON I.VENDORID = V.VENDORID
INNER JOIN Products P
    ON I.PRODUCTSID = P.PRODUCTSID
WHERE I.QUANTITYAVAILABLE < 0   -- Low stock threshold
ORDER BY V.VENDORNAME, I.QUANTITYAVAILABLE ASC;

--10 SEARCH PRODUCTS BY KEYWORD (FLEXIBLE QUERY)

14. Security & RoleBased Access Control

create login vendorlogin with password = 'mohsn@123'

--map login to data user 
use moh;
create user admin_user  for login adminlogin;
create user Vendor_user  for login Vendorlogin;

--3 create role
create role adminRole;
create role VendorRole;

--4 assign user to roles
alter role Adminrole Add member admin_user ;
exec sp_addrolemember 'VendorRole','Vendor_user';

--5 Grant /revoke permisssion to role
